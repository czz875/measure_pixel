    import argparse
    parser = argparse.ArgumentParser(description="图片测量工具")
    parser.add_argument("--project", help="加载已有项目 .project.json")
    parser.add_argument("--folder", help="启动时加载文件夹内的所有图片")
    args = parser.parse_args()

    root = tk.Tk()
    app = ImageMeasureApp(root, style_file="ui_styles.json")
    if args.project:
        try:
            import json
            with open(args.project, "r", encoding="utf-8") as f:
                project_data = json.load(f)
            from src.core.commands import load_project_data
            load_project_data(app, project_data)
        except Exception as e:
            print(f"Failed to load project: {e}")
    if args.folder:
        from src.persistence.file_handler import FileHandler
        images = FileHandler.scan_image_folder(args.folder)
        app.image_paths = images
        if images:
            app.index = 0
            app.load_image()
    root.mainloop()
